class Monster {
  constructor(name, emoji, img, hpMin, hpMax, gold, exp, boss = false, element = null) {

      'earth': 'wind', 'thunder': 'earth', 'water': 'thunder',
      'light': 'dark', 'dark': 'light'
    };
    return weaknesses[attackElement] === this.element;
  }
}

### 6. Combo Enhancement (`src/engine/GameEngine.js`)

**Current:** Simple combo counter

**Enhanced formula per SPEC.md:**
```javascript
calculateDamageWithCombo(baseDamage, answerTime, combo) {
  // Time bonus
  let timeBonus = 1.0;
  if (answerTime <= 5) timeBonus = 1.5;
  else if (answerTime <= 10) timeBonus = 1.2;
  else if (answerTime <= 20) timeBonus = 1.0;
  else timeBonus = 0.8;

  // Combo multiplier
  let comboMultiplier = 1.0;
  if (combo >= 5) comboMultiplier = 1.1;
  if (combo >= 10) comboMultiplier = 1.25;
  if (combo >= 20) comboMultiplier = 1.5;
  if (combo >= 50) comboMultiplier = 2.0;

  // Class bonus
  const hero = HEROES[this.state.heroKey];
  const classBonus = this.getClassDamageBonus(hero.element);

  return Math.floor(baseDamage * timeBonus * comboMultiplier * classBonus);
}

getClassDamageBonus(element) {
  const bonuses = { mage: 1.15, knight: 1.10, berserker: 1.20 };
  return bonuses[this.state.heroKey] || 1.0;
}
```
