### 3.3 Add `selectInputLevel(level)` Function

**Location**: After `selectStage()` function (around line 5755)

```javascript
let selectedInputLevel = null; // 1, 2, or 3

function selectInputLevel(level) {
  selectedInputLevel = level;
  // Update visual selection
  document.querySelectorAll('.level-card').forEach(card => {
    card.classList.toggle('selected', parseInt(card.dataset.level) === level);
  });
  // Persist
  localStorage.setItem('selectedInputLevel', level);
}

function showLevelSelect() {
  // Hide stage selector, show level selector
  document.getElementById('stageSelector').style.display = 'none';
  document.getElementById('battleStartBtnOverlay').style.display = 'none';
  document.getElementById('levelSelectScreen').style.display = 'block';
  // Load saved level
  const saved = localStorage.getItem('selectedInputLevel');
  if (saved) selectInputLevel(parseInt(saved));
}

function showStageSelector() {
  if (!selectedInputLevel) {
    alert('請先選擇答題難度');
    return;
  }
  document.getElementById('levelSelectScreen').style.display = 'none';
  document.getElementById('stageSelector').style.display = 'block';
  document.getElementById('battleStartBtnOverlay').style.display = 'block';
  renderStageSelector();
}
```
